radarData = [
    {axis: "Energy", value: currentTrack.energy },
    {axis: "Dance", value: currentTrack.danceability },
    {axis: "Acoustic", value: currentTrack.acousticness },
    {axis: "Valence", value: currentTrack.valence },
    {axis: "Speech", value: currentTrack.speechiness }
];

{
    // 1. Aumentamos as dimensões e a margem (espaço para o texto)
    const width = 300; 
    const height = 300;
    const margin = 50; // Margem generosa para as labels
    
    const radius = Math.min(width, height) / 2 - margin;
    
    // Criar o SVG com um viewBox que abrange tudo
    const svg = d3.create("svg")
        .attr("viewBox", [0, 0, width, height])
        .style("overflow", "visible");

    const g = svg.append("g")
        .attr("transform", `translate(${width/2},${height/2})`);
    
    const angleSlice = (Math.PI * 2) / radarData.length;
    const rScale = d3.scaleLinear().domain([0, 1]).range([0, radius]);

    // 2. Desenhar as linhas dos eixos
    const axis = g.selectAll(".axis")
        .data(radarData)
        .enter().append("g");

    axis.append("line")
        .attr("x1", 0)
        .attr("y1", 0)
        .attr("x2", (d, i) => rScale(1) * Math.cos(angleSlice * i - Math.PI/2))
        .attr("y2", (d, i) => rScale(1) * Math.sin(angleSlice * i - Math.PI/2))
        .style("stroke", "#999")
        .style("stroke-width", "1px");

    // 3. Adicionar Labels (Com coordenadas explícitas e cor visível)
    axis.append("text")
        .attr("x", (d, i) => rScale(1.25) * Math.cos(angleSlice * i - Math.PI/2))
        .attr("y", (d, i) => rScale(1.25) * Math.sin(angleSlice * i - Math.PI/2))
        .text(d => d.axis)
        .style("font-family", "sans-serif")
        .style("font-size", "16px") // Texto maior
        .style("font-weight", "bold")
        .style("fill", "#0be7d5") // Usar o ciano da tua marca para garantir visibilidade
        .attr("text-anchor", "middle")
        .attr("dy", "0.35em");

    // 4. Desenhar os níveis (círculos tracejados)
    g.selectAll(".grid-circle")
        .data([0.2, 0.4, 0.6, 0.8, 1])
        .enter().append("circle")
        .attr("r", d => rScale(d))
        .style("fill", "none")
        .style("stroke", "#444")
        .style("stroke-dasharray", "4,4");

    // 5. Desenhar o polígono (Área preenchida)
    const radarLine = d3.lineRadial()
        .curve(d3.curveLinearClosed)
        .radius(d => rScale(d.value))
        .angle((d, i) => i * angleSlice);

    g.append("path")
        .datum(radarData)
        .attr("d", radarLine)
        .style("fill", currentTrack.category === "High" ? "#0be7d5" : "#ff4d4d")
        .style("fill-opacity", 0.5)
        .style("stroke", "white")
        .style("stroke-width", "2px");

    return svg.node();
}

{
    // 1. Definição do URL (usando a lógica de fallback que tinhas)
    const audioURL = `https://p.scdn.co/mp3-preview/${selectedTrack.track_id}`;

    // 2. Criação do elemento de áudio
    const audio = html`<audio crossorigin="anonymous" controls src="${audioURL}" style="width: 100%; filter: invert(1);"></audio>`;

    // 3. Configuração do Analisador (Web Audio API)
    const ctx = new (window.AudioContext || window.webkitAudioContext)();
    const analyser = ctx.createAnalyser();
    
    // CORREÇÃO: Usar o elemento 'audio' que acabaste de criar
    const source = ctx.createMediaElementSource(audio);
    source.connect(analyser);
    analyser.connect(ctx.destination);
    
    analyser.fftSize = 1024;
    const bufferLength = analyser.frequencyBinCount;
    const dataArray = new Uint8Array(bufferLength);

    // 4. Criação do Canvas
    const canvas = html`<canvas width="400" height="120" style="width: 100%; background: #1a1a1a; border-radius: 8px; margin-top: 10px;"></canvas>`;
    const canvasCtx = canvas.getContext("2d");

    // 5. Função de desenho corrigida
    function draw() {
        requestAnimationFrame(draw);
        analyser.getByteTimeDomainData(dataArray);

        // CORREÇÃO: Usar canvasCtx e não ctx (ctx é o áudio!)
        canvasCtx.fillStyle = "#1a1a1a";
        canvasCtx.fillRect(0, 0, canvas.width, canvas.height);
        
        canvasCtx.lineWidth = 2;
        canvasCtx.strokeStyle = "#0be7d5";
        canvasCtx.beginPath();

        const sliceWidth = canvas.width / bufferLength;
        let x = 0;

        for (let i = 0; i < bufferLength; i++) {
            const v = dataArray[i] / 128.0;
            const y = (v * canvas.height) / 2;

            if (i === 0) canvasCtx.moveTo(x, y);
            else canvasCtx.lineTo(x, y);

            x += sliceWidth;
        }

        canvasCtx.lineTo(canvas.width, canvas.height / 2);
        canvasCtx.stroke();
    }

    draw();

    // 6. Retorno do HTML
    return html`
        <div style="display: flex; flex-direction: column; gap: 10px; padding: 15px; background: #222; border-radius: 12px; border: 1px solid #333;">
            <h6 style="margin:0; color:#888; font-size: 0.7em; text-transform: uppercase;">A analisar sinal de:</h6>
            <div style="font-weight: bold; color: white; margin-bottom: 5px;">${selectedTrack?.track_name || "Selecione uma música"}</div>

            ${audio}
            ${canvas}

            <div style="margin-top: 10px; padding: 10px; background: #1a1a1a; border-radius: 8px; border-left: 4px solid #0be7d5;">
                <span style="font-size: 0.8em; color: #aaa;">Visualização em tempo real do domínio do tempo (Waveform).</span>
            </div>
        </div>
    `;
}
